\question[8]
SAMPLE QUESTION. Consider two conditions on languages:
%
\begin{itemize}
    \item[C1:] Contains at least a word containing ``$ab$'' as a substring and ending in $b$.
    \item[C2:] Contains at least a word of the form $a^kb^kb$ for some $k \in \naturals$.
\end{itemize}

For each of the following regular expressions, indicate whether the language
generated by it satisfies only C1, only C2, both, or neither:

\begin{parts}
    % each part is a regex
    \part \(a^*b^*b\)

    \begin{oneparchoices}
        \choice Only C1
        \choice Only C2
        \CorrectChoice Both
        \choice Neither
    \end{oneparchoices}
    %
    \part \((ab)^*\)

    \begin{oneparchoices}
        \CorrectChoice Only C1
        \choice Only C2
        \choice Both
        \choice Neither
    \end{oneparchoices}
    %
    \part \((a \mid b)^*\)

    \begin{oneparchoices}
        \choice Only C1
        \choice Only C2
        \CorrectChoice Both
        \choice Neither
    \end{oneparchoices}
    %
    \part \((a \mid b)^* a (a \mid b)^* b\)

    \begin{oneparchoices}
        \choice Only C1
        \choice Only C2
        \CorrectChoice Both
        \choice Neither
    \end{oneparchoices}
\end{parts}
